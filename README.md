## Install

```
npm i remark github:3846masa/remark-review#latest
```

## Usage

```js
const remark = require('remark');
const review = require('@3846masa/remark-review');

const markdown = `
# Introduction

This paper proposes ...
`;

remark().use(review).process(markdown, (err, vfile) => {
  if (err) {
    console.error(err);
  }
  console.log(vfile);
});
```
